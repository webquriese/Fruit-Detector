    private  BufferedImage resizeImage(BufferedImage originalImage, int type)
    {
                      int IMG_WIDTH=250,IMG_HEIGHT=250;  
	BufferedImage resizedImage = new BufferedImage(IMG_WIDTH, IMG_HEIGHT, type);
	Graphics2D g = resizedImage.createGraphics();
	g.drawImage(originalImage, 0, 0, IMG_WIDTH, IMG_HEIGHT, null);
	g.dispose();
 
	return resizedImage;
    }

    private void jButton3ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        
        try
        {
            int k = Integer.parseInt( JOptionPane.showInputDialog(this, "Enter No Of Clusters", "3"));
            String reply = ClientUtil.sendMsg("receivekmeans:"+colorx+":"+colory+":"+k);
            
            if ( reply.equalsIgnoreCase("success"))
            {
                kk++;
                ClientUtil.sendFile(f);
                reply = ClientUtil.sendMsg("send");
                System.out.println("Got "+reply);
                if ( reply.equalsIgnoreCase("success"))
                {
                    ClientUtil.recvFile(new File("temp"+kk+".jpg"));
                    
                    reply=ClientUtil.sendMsg("cluster");
                    if ( reply.equalsIgnoreCase("NA"))
                        JOptionPane.showMessageDialog(null,"Error");
                    else
                    {
                        double i = Double.parseDouble(reply.trim());
                        System.out.println(i);
                        String msg = "";
                        if ( i > 30)
                            msg = " Above 30 %";
                        else  if ( i > 20 && i < 30)
                            msg = " Above 20 % Below 30%";
                        else
                            msg = "Below 20 %";
                        
                        JOptionPane.showMessageDialog(null,msg);
                    }
                    jLabel1.setIcon(new ImageIcon("temp"+kk+".jpg"));
                }
                    
            }
        }
        catch(Exception e)
        {
            /*e.printStackTrace();*/
        }
    }                                        

    private void jLabel1MouseClicked(java.awt.event.MouseEvent evt) {                                     

        try
        {
            int x1 = evt.getX();
            int y1 = evt.getY();            
            BufferedImage image = ImageIO.read(f);
            w = image.getWidth(); 
            h = image.getHeight();         
            for (int y=0;y<h;y++) { 
                for (int x=0;x<w;x++) 
                { 
                    int rgb = image.getRGB(x, y); 
                    int r = rgb>>16&0x000000FF;  
                    int g = rgb>> 8&0x000000FF;  
                    int b = rgb&0x000000FF;  
                    if ( x1 == x && y1 == y)
                    {
                        colorx = x;
                        colory = y;
                        txtR.setText(""+r);
                        txtGreen.setText(""+g);
                        txtBlue.setText(""+b);
                        System.out.println("------------------------------Red"+r + " Green " + g + "Blue " + b); 
                    }
                }
            }
    
        }
        catch(Exception e)
        {
            /*e.printStackTrace();*/
        }
    }
